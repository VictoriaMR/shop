  `site_id` tinyint(4) NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `path` varchar(32) NOT NULL DEFAULT '' COMMENT '模板',
  `name` varchar(64) NOT NULL DEFAULT '' COMMENT '名称',
  `status` tinyint(1) NOT NULL DEFAULT '0' COMMENT '状态',
  `keyword` varchar(255) NOT NULL DEFAULT '' COMMENT 'keyword关键字',
  `description` varchar(255) NOT NULL DEFAULT '' COMMENT 'desc关键字',
  `remark` varchar(255) NOT NULL DEFAULT '' COMMENT '内部使用备注',
  `add_time` timestamp NULL DEFAULT NULL COMMENT '加入时间',
  `update_time` timestamp NULL DEFAULT NULL COMMENT '更新时间',
  PRIMARY KEY (`site_id`)
) ENGINE=InnoDB AUTO_INCREMENT=81 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
